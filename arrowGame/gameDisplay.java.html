    public void startGame() {
        int downPositionX = downArrowTransparent.getX();

    public void pauseGame(){
        pause = !pause;
        if (pause) {
            myTimer.cancel();
        } else {
            myTimer = new Timer();
            moveArrows();
        }
    }

    public void score() {
        if ((score%50 == 0) && (score > 100)) {
            if (score > 100) {
                max = (score / 50) + 3;
                min = (score / 50) -1;
            }
        }
        if (score > -1) {
            highScoreDisplay.setText("Highscore: " + highestScore);
            scoreDisplay.setText("Score: " + score);
            if (score < 100) {
                backgroundScore.setText("0");
                backgroundScoreTwo.setText(String.valueOf(score));
            } else {
                char[] scoreArr = String.valueOf(score).toCharArray();
                backgroundScore.setText(String.valueOf(scoreArr[0]));
                backgroundScoreTwo.setText(String.valueOf(scoreArr[1])+String.valueOf(scoreArr[2]));
            }
        } else {
            endGame();
        }
    }
    public void checkHighestScore(){
        if (score > highestScore){
            highestScore = score;
        }
    }
    public void checkLocation(){
        int deadZoneY = deadZone.getY();
        if (downArrow.getY() > deadZoneY){
            downArrow.setLocation(downArrow.getX(), -200);
            checkHighestScore();
            score--;
            score();
        } else if (upArrow.getY() > deadZoneY){
            upArrow.setLocation(upArrow.getX(), -200);
            checkHighestScore();
            score--;
            score();
        } else if (leftArrow.getY() > deadZoneY){
            leftArrow.setLocation(leftArrow.getX(), -200);
            checkHighestScore();
            score--;
            score();
        } else if (rightArrow.getY() > deadZoneY) {
            rightArrow.setLocation(rightArrow.getX(), -200);
            checkHighestScore();
            score--;
            score();
        }
    }
    public void moveArrows() {
        myTimer.schedule(new TimerTask() {
            public void run()  {
                checkLocation();
                int downPositionX = downArrowTransparent.getX();
                int upPositionX = upArrowTransparent.getX();
                int leftPositionX = leftArrowTransparent.getX();
                int rightPositionX = rightArrowTransparent.getX();
                int downPositionY = downArrow.getY();
                int upPositionY = upArrow.getY();
                int leftPositionY = leftArrow.getY();
                int rightPositionY = rightArrow.getY();
                downArrow.setLocation(downPositionX, downPositionY+randomNumDown);
                upArrow.setLocation(upPositionX, upPositionY+randomNumUp);
                leftArrow.setLocation(leftPositionX, leftPositionY+randomNumLeft);
                rightArrow.setLocation(rightPositionX, rightPositionY+randomNumRight);
            }
        }, 1, 5);
    }

class myListener implements KeyListener {

    @Override
    public void keyTyped(KeyEvent e) {

    }

    @Override
    public void keyPressed(KeyEvent e) {
        int downArrowTransparentY = downArrowTransparent.getY();
        int upArrowTransparentY = upArrowTransparent.getY();
        int leftArrowTransparentY = leftArrowTransparent.getY();
        int rightArrowTransparentY = rightArrowTransparent.getY();
        switch (e.getKeyCode()) {
            case KeyEvent.VK_DOWN:
            case KeyEvent.VK_S:
                if (downArrow.getY() > downArrowTransparentY - 100 && downArrow.getY() < downArrowTransparentY + 100) {
                    downArrow.setLocation(downArrow.getX(), -200);
                    randomNumDown = (int) (Math.random() * (max - min + 1) + min);
                    score++;
                    checkHighestScore();
                } else {
                    checkHighestScore();
                    score--;
                }
                break;
            case KeyEvent.VK_UP:
            case KeyEvent.VK_W:
                if (upArrow.getY() > upArrowTransparentY - 100 && upArrow.getY() < upArrowTransparentY + 100) {
                    upArrow.setLocation(upArrow.getX(), -200);
                    randomNumUp = (int) (Math.random() * (max - min + 1) + min);
                    score++;
                    checkHighestScore();
                } else {
                    checkHighestScore();
                    score--;
                }
                break;
            case KeyEvent.VK_LEFT:
            case KeyEvent.VK_A:
                if (leftArrow.getY() > leftArrowTransparentY - 100 && leftArrow.getY() < leftArrowTransparentY + 100) {
                    leftArrow.setLocation(leftArrow.getX(), -200);
                    randomNumLeft = (int) (Math.random() * (max - min + 1) + min);
                    score++;
                    checkHighestScore();
                } else {
                    checkHighestScore();
                    score--;
                }
                break;
            case KeyEvent.VK_RIGHT:
            case KeyEvent.VK_D:
                if (rightArrow.getY() > rightArrowTransparentY - 100 && rightArrow.getY() < rightArrowTransparentY + 100) {
                    rightArrow.setLocation(rightArrow.getX(), -200);
                    randomNumRight = (int) (Math.random() * (max - min + 1) + min);
                    score++;
                    checkHighestScore();
                } else {
                    checkHighestScore();
                    score--;
                }
                break;
            case KeyEvent.VK_ESCAPE:
                pauseGame();
        }
        score();
    }
    @Override
    public void keyReleased(KeyEvent e) {
    }
}
